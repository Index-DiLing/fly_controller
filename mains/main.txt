
#include "stm32f4xx.h"
#include "dl_gpio.hpp"
#include "dl_nvic_it.h"
#include "dl_delay.hpp"
#include "dl_socket.hpp"
#include "dl_usart.hpp"

#include "global.h"

#define MAX_16_BITS 65536

#define CODE_END    while (1);
inline void GPIOClockInit()
{

    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOA, ENABLE);
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOF, ENABLE);
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOB, ENABLE);
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOC, ENABLE);
}

inline void init()
{
    GPIOClockInit();

    NVIC_PriorityGroupConfig(NVIC_PriorityGroup_2);
    SysTick_Config(SystemCoreClock / 1000);

    dl::delay_ms(100);
}
// TX:PA9

static uint8_t rx_buf[128];
/**
 * @brief 占用IO:
 *
 * 接线:
 * PA11 - Debug LED接灯
 * PE9  TIM  CH1  接信号线176
 * PE11 TIM  CH2
 * PE13 TIM  CH3
 * PE14 TIM  CH4
 *
 * 电调负极线与单片机负极线相连
 *
 * @return int
 */
#include "dl_log.h"
#include "dl_W25N01G.hpp"
int main()
{

    init();
    dl::USART usart1X(dUSART1_tA9_rA10, rx_buf, sizeof(rx_buf), 1, 1, 1520000);
    dl::Socket socket(usart1X);
    log_func = [&socket](const char *msg) {
        socket.sendString(msg);
    };
    logger << "\r\n"
           << LCMD::FLUSH;
    logger << "Program Started: \n"
           << LCMD::FLUSH;

    dl::GPIO_Pin_Init(dGPIOFPin3, GPIO_Mode_OUT, GPIO_Speed_50MHz, GPIO_OType_PP, GPIO_PuPd_NOPULL);
    dl::GPIO_Pin_Init(dGPIOFPin4, GPIO_Mode_OUT, GPIO_Speed_50MHz, GPIO_OType_PP, GPIO_PuPd_NOPULL);
    dl::GPIO_Pin_Init(dGPIOFPin5, GPIO_Mode_OUT, GPIO_Speed_50MHz, GPIO_OType_PP, GPIO_PuPd_NOPULL);

    dPF3 = 1;
    dPF4 = 1;
    dPF5 = 1;
    dl::delay_ms(100);
    dPF3 = 0;

    uint8_t data[1024];
    dl::SPI xspi1(dSPI1_SA5_MIA6_MOA7, dGPIOCPin8);

    dl::W25N01ZEIG nand(xspi1);

    nand.disableWriteProtect();

    nand.writeRegister(dl::W25N01ZEIG_STATUS_REGISTER::STATUS, 0x00);
    
    logger << "STATUS:" << nand.readStatusRegister(dl::W25N01ZEIG_STATUS_REGISTER::STATUS) << LCMD::NFLUSH;
    dl::delay_ms(100);

    
    nand.writeEnable();
    nand.erase(156);

    uint8_t *str = (uint8_t *)"ABCDEFGHIJKL";
    nand.writeEnable();


    nand.loadProgramData(0, str, 12);

    dl::delay_ms(10);

    nand.readData(0, &(data[0]), 12);

    for (uint16_t i = 0; i < 12; i++) {
        logger << data[i] << " ";
    }
    logger << LCMD::NFLUSH;

    logger << "PROTECTION:" << nand.readStatusRegister(dl::W25N01ZEIG_STATUS_REGISTER::PROTECTION) << LCMD::NFLUSH;
    dl::delay_ms(100);
    nand.writeEnable();
    nand.programExecute(156);

    logger << "STATUS:" << nand.readStatusRegister(dl::W25N01ZEIG_STATUS_REGISTER::STATUS) << LCMD::NFLUSH;
    dl::delay_ms(1000);
    nand.pageDataRead(156);
    dl::delay_ms(100);
    nand.readData(0, &(data[0]), 13);

    for (uint16_t i = 0; i < 13; i++) {
        logger << data[i] << " ";
    }
    logger << LCMD::NFLUSH;

    while (true) {
    }
}
